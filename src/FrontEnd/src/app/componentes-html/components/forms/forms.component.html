DatePicker Theme

Datepicker viene con algunos temas de color predeterminados. Puede cambiarlo manipulando la propiedad containerClass en el objeto bsConfig

Actualmente cuenta con 6 temas para el color: theme-default, theme-green, theme-blue, theme-dark-blue, theme-red, theme-orange 

                      
                        <div class="col-md-6 form-group">
                          <label class="form-label">Fecha</label>
                          <input
                            type="text"
                            placeholder="Datepicker"
                            class="form-control"
                            bsDatepicker
                            [bsConfig]="bsConfig"
                            #dp="bsDatepicker"
                          >
                        </div>
                        <div class="col-md-6 form-group">
                          <label class="form-label">Color de Tema:</label>
                          <select
                             class="custom-select form-control from-select" 
                             [(ngModel)]="colorTheme" 
                             (ngModelChange)="applyTheme(dp);" 
                          >
                             <option value="theme-default">default</option>
                             <option value="theme-green">green</option>
                             <option value="theme-blue">blue</option>
                             <option value="theme-dark-blue">dark-blue</option>
                             <option value="theme-red">red</option>
                             <option value="theme-orange">orange</option>
                          </select>
                        </div>
                      
                    
                      
                        import { Component  } from '@angular/core';
                        import { BsDatepickerConfig   } from 'ngx-bootstrap/datepicker';

                        @Component({

                        )}

                        export class DemoDataPickerThemeComponent {

                         colorTheme="theme-green";
                         bsConfig?=Partial<BsDatepickerConfig>;

                         applyTheme(pop: any) {
                        // crea un nuevo objeto en cada cambio de propiedad
                        // para que Angular pueda detectar el cambio de referencia del objeto
                           this.bsConfig = Object.assign({}, { containerClass: this.colorTheme });
                           setTimeout(() => {
                             pop.show();
                           });
                         }
                        }

DatePicker con Formato Fecha

Datepicker viene con el formato MM-DD-YYYY. Puede cambiarlo manipulando la propiedad dateInputFormat en el objeto bsConfig establezca el formato DD/MM/YYYY.
Se recomienda establecerlo en el documento TypeScript y enviarle la variable a la propiedad bsConfig

                      <div class="col-md-6 form-group">
                          <label>Fecha</label>
                          <input
                            type="text"
                            placeholder="Datepicker"
                            class="form-control"
                            bsDatepicker
                            [bsConfig]={ dateInputFormat: 'DD/MM/YYYY' }
                          >
                        </div>
                      
                    
                      import { Component  } from '@angular/core';
                        import { BsDatepickerConfig   } from 'ngx-bootstrap/datepicker';

                        @Component({

                        )}

                        export class DemoDataPickerFormatComponent {

                         colorTheme="theme-green";
                         bsConfig?=Partial<BsDatepickerConfig> = Object.assign({}, { dateInputFormat: 'DD/MM/YYYY'})  ;

                        }

Input File

                            accept="image/png"
                            [required]="true"

                          save(file : any ) {

                          }

Input File

Use el componente <file-upload></file-upload> y agregue el atributo [multiple]="true" para cargar multiples archivo

                        
                          <file-upload
                            accept="image/png"
                            [required]="true"
                            (model)="save($event)"
                            [multiple]="true"
                            classBtnUpload="btn-file-upload"
                            colorModalViewFile="danger"
                          ></file-upload>
                        
                      
                        
                          save(file : any ) {
                            console.log("fileModel", file) // retorna un array de archivos 
                          }

Input File - Acepta multiples tipos de archivos

Agregue el atributo accept="application/pdf, image/png" entre comillas, todos los tipos de archivos que desea agregar, separados por una coma

                        
                          <file-upload
                            accept="application/pdf, image/png"
                            [required]="true"
                            (model)="save($event)"
                            [multiple]="true"
                            classBtnUpload="btn-file-upload"
                            colorModalViewFile="danger"
                          ></file-upload>
                        
                      
                        
                          save(file : any ) {
                            console.log("fileModel", file) // retorna un array de archivos 
                          }
