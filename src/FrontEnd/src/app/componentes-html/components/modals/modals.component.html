Modal Básica

Use la clase modal-content para crear una ventana modal

<div class="modal-content">
 <div class="modal-header">
   <h4 class="modal-title pull-left">Modal Básica</h4>
     <button aria-label="Close" data-bs-dismiss="modal" class="btn-close">
       <span aria-hidden="true">x</span>
     </button>
   </div>
 <div class="modal-body">
   <h5>Este es el cuerpo del Modal </h5>
   <p>Lorem, ipsum dolor sit amet consectetur adipisicing elit. Illum sed maxime mollitia 
 iusto voluptatibus distinctio deserunt expedita at ipsa enim suscipit fugiat ducimus soluta,
delectus quis dolorem, modi dolorum doloribus..</p>
 </div>
 <div class="modal-footer">
   <button class="btn btn-primary">Guardar Cambios</button>
   <button class="btn btn-light">Cancelar</button>
 </div>
</div>
// Demo
openModal(template: TemplateRef<any> {
 this.modalRef = this.modalService.show(template);
}
// Modal Static
openModal(template: TemplateRef<any> {
 this.modalRef = this.modalService.show(template. { backdrop: 'static'});
}
// Modal Static Center
openModal(template: TemplateRef<any> {
 this.modalRef = this.modalService.show(template. { backdrop: 'static', class: 'modal-dialog-centered' });
}

Modal Personalizada

Use las clases modal-content modal-minfin seguido de la variacion de color (primary, secondary, danger, etc.) deseada, para crear una ventana modal personalizada

<div class="modal-content modal-minfin danger">
 <div class="modal-header">
   <h4 class="modal-title pull-left">Modal Básica</h4>
     <button aria-label="Close" data-bs-dismiss="modal" class="btn-close">
       <span aria-hidden="true">x</span>
     </button>
   </div>
 <div class="modal-body">
   <p>Lorem, ipsum dolor sit amet consectetur adipisicing elit. Illum sed maxime mollitia 
 iusto voluptatibus distinctio deserunt expedita at ipsa enim suscipit fugiat ducimus soluta,
delectus quis dolorem, modi dolorum doloribus..</p>
 </div>
 <div class="modal-footer">
   <button class="btn btn-primary">Guardar Cambios</button>
   <button class="btn btn-light">Cancelar</button>
 </div>
</div>
// Modal Personalizada
openModal(template: TemplateRef<any> {
 this.modalRef = this.modalService.show(template);
}

Modal Personalizada Simple

Use las clases modal-content modal-minfin sin la etiqueta modal-header 

<div class="modal-content modal-minfin">

   <p>Lorem, ipsum dolor sit amet consectetur adipisicing elit. Illum sed maxime mollitia

openModal(template: TemplateRef<any> {

Modal Personalizada FullScreen

Use las clases modal-content modal-minfin seguido de la variacion de color (primary, secondary, danger, etc.) deseada, para crear una ventana modal personalizada y en el archivo .ts envie el parametro class: 'modal-fullscreen' 

<div class="modal-content modal-minfin danger">
 <div class="modal-body">
   <h4 class="modal-title mb-2">Modal Personalizada Simple</h4>
   <p>Lorem, ipsum dolor sit amet consectetur adipisicing elit. Illum sed maxime mollitia 
 iusto voluptatibus distinctio deserunt expedita at ipsa enim suscipit fugiat ducimus soluta,
delectus quis dolorem, modi dolorum doloribus..</p>
 </div>
 <div class="modal-footer">
   <button class="btn btn-primary">Guardar Cambios</button>
   <button class="btn btn-light">Cancelar</button>
 </div>
</div>
// Modal Personalizada
openModal(template: TemplateRef<any> {
 this.modalRef = this.modalService.show(template, { class: 'modal-fullscreen' });
}

Modal Tamaños

En el archivo .ts envie en el parametro class: las clases 'modal-sm' | 'modal-md' | 'modal-lg' | 'modal-xl'
